# Fonksiyonlarda bazı çılgınlıklar
var funct1 = function fun1(){console.Log("function 1")}
// fun1() fonksiyonunu direkt olarak çağıramam
funct1() --> function 1

const funct2 = () => {console.Log("function 2")}
// buradaki "()" işareti bir fonksiyonu belirtiyor "=>" işareti de lambda gösterimi tarzı bir şey
funct2() --> function 2

# map işlemi
var list = [1,2,3,4,5]

var squareMap = list.map(number => {
    number*number
})
console.Log(squareMap) --> 1,3,9,16,25

// bu işlemi aşağıdaki yolla da yapabilriz
var squareList = []
list.forEach(number => squareList.push(number*number))
console.Log(squareList) --> 1,3,9,16,25

# filter -> dizideki elemanlara filtreleme yapabilmeyi sağlıyor
var list = [2,5,21,14,32]
const filterList = list.filter(number => number>4) 
console.Log(filterList) --> [5,21,14,32]
